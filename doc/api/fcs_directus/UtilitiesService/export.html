Future<dynamic> export(
  String collection, {
  String format = 'json',
  Map<String, dynamic>? query,
}) async {
  return await _httpClient.get(
    '/utils/export/$collection',
    queryParameters: {'format': format, if (query != null) ...query},
  );
}
