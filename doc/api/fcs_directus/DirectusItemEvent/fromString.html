static DirectusItemEvent? fromString(String? value) {
  if (value == null) return null;
  switch (value.toLowerCase()) {
    case 'create':
      return DirectusItemEvent.create;
    case 'update':
      return DirectusItemEvent.update;
    case 'delete':
      return DirectusItemEvent.delete;
    default:
      return null;
  }
}
